def compute_gradient(X, y, theta):
    diff = np.dot(X, theta) - y
    return (1 / len(X)) * np.dot(X.T, diff)

def gradient_descent(X, y, alpha, theta):
    history_cost = []
    
    while True:
        history_cost.append(compute_cost(X, y, theta))
        gradient = compute_gradient(X, y, theta)
        theta = theta - alpha * gradient
        
         # 梯度小于 1e-5 则判定取得极小值
        if(np.all(np.absolute(gradient) <= np.exp(-5))):
            break
        
    return theta, history_cost

# 正规方程
def normalEqn(X, y):
    # linalg 模块包含线性代数的函数，inv() 函数计算逆矩阵
    theta = np.linalg.inv(X.T@X)@X.T@y    #X.T@X 等价于 X.T.dot(X)
    return theta
